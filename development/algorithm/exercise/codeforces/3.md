# A

```ruby
s = gets
p1 = s[0].ord - 97
p2 = s[1].to_i - 1

t = gets
q1 = t[0].ord - 97
q2 = t[1].to_i - 1

puts [(p1 - q1).abs, (p2 - q2).abs].max

while (p1 != q1 || p2 != q2) 
    d = ''
    if p1 < q1
        d += 'R'
        p1 += 1
    elsif p1 > q1
        d += 'L'
        p1 -= 1
    end
    if p2 < q2
        d += 'U'
        p2 += 1
    elsif p2 > q2
        d += 'D'
        p2 -= 1
    end
    puts d
end
```
