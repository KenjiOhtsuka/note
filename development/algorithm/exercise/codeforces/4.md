        int diff = min(h[i].cap, sumTime - sumMin);
 
        h[i].minTime += diff;
        h[i].cap -= diff;
        sumMin += diff;
 
        cout << h[i].minTime << ' ';
    }

}

fun readString(): String {
    return readLine()!!.trim()
}

fun readIntList(): List<Int> {
    return readLine()!!.split(' ').map { it.toInt() }
}

    val n = readInt()

    val map = mutableMapOf<String, Int>()

        val num: Int
