```c
#include <stdio.h>

int main() {
    int a, b;
    while (scanf("%d %d", &a, &b) != EOF) {
        int ans = 1;

        int c = a + b;
        for (int register i = 0;; i++) {
            if (c > 0) ans += 1;
            else break;
        }
        printf("%d\n", ans);
    }
}
```
